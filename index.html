Step 1 Open sheets.new to create a new Google Sheet. Name it BESS Issue Tracker
Step 2 In the Sheet menu → Extensions → Apps Script. Delete any code in the editor, then paste this entire script:
// Paste in Apps Script editor — see downloaded file // "backend_script.gs" for the full code // Or copy from below: const SHEET_NAME = "Issues"; const COLUMNS = ["id","project","device","customDevice","alarm","srNo","occurredDate","resolvedDate","task","taskAssignee","taskStatus","manhours","siteMaintenanceStatus","officeMaintenanceStatus","remark"]; function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.getRange(1,1,1,COLUMNS.length).setValues([COLUMNS]).setFontWeight("bold"); sheet.setFrozenRows(1); } const data = sheet.getDataRange().getValues(); if (data.length <= 1) return cr({success:true,issues:[]}); const h=data[0], issues=[]; for (let i=1;i<data.length;i++) { const r=data[i]; if(!r[0]&&!r[4])continue; const o={}; h.forEach((k,j)=>{o[k]=r[j]||""}); o.id=Number(o.id)||i; issues.push(o); } return cr({success:true,issues}); } catch(err) { return cr({success:false,error:err.message}); } } function doPost(e) { try { const body=JSON.parse(e.postData.contents), action=body.action; const ss=SpreadsheetApp.getActiveSpreadsheet(); let sheet=ss.getSheetByName(SHEET_NAME); if(!sheet){sheet=ss.insertSheet(SHEET_NAME);sheet.getRange(1,1,1,COLUMNS.length).setValues([COLUMNS]).setFontWeight("bold");sheet.setFrozenRows(1);} if(action==="save_all"){ const issues=body.issues||[], lr=sheet.getLastRow(); if(lr>1) sheet.getRange(2,1,lr-1,COLUMNS.length).clear(); if(issues.length>0){const rows=issues.map(i=>COLUMNS.map(c=>i[c]||"")); sheet.getRange(2,1,rows.length,COLUMNS.length).setValues(rows);} return cr({success:true,count:issues.length}); } else if(action==="add"){ const i=body.issue; sheet.appendRow(COLUMNS.map(c=>i[c]||"")); return cr({success:true,id:i.id}); } else if(action==="update"){ const i=body.issue, data=sheet.getDataRange().getValues(); for(let r=1;r<data.length;r++){if(Number(data[r][0])===Number(i.id)){sheet.getRange(r+1,1,1,COLUMNS.length).setValues([COLUMNS.map(c=>i[c]||"")]);return cr({success:true,id:i.id});}} sheet.appendRow(COLUMNS.map(c=>i[c]||"")); return cr({success:true,id:i.id,appended:true}); } else if(action==="delete"){ const id=Number(body.id), data=sheet.getDataRange().getValues(); for(let r=data.length-1;r>=1;r--){if(Number(data[r][0])===id){sheet.deleteRow(r+1);return cr({success:true,deleted:id});}} } else if(action==="clear_all"){ const lr=sheet.getLastRow(); if(lr>1)sheet.getRange(2,1,lr-1,COLUMNS.length).clear(); return cr({success:true}); } return cr({success:false,error:"Unknown action"}); } catch(err) { return cr({success:false,error:err.message}); } } function cr(o){return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON);}
